const handleDonate = async (amount) => {
  const response = await fetch(
    `${import.meta.env.VITE_API_URL}/api/donations`,
    {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        campaignId: campaign.id,
        amount: amount,
        method: "flutterwave",
      }),
    },
  );
  const data = await response.json();
